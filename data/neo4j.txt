# 导入entity
LOAD CSV WITH HEADERS FROM "file:///entity.csv" AS line CREATE (p:EntityItem{title:line.title})

# 导入new_node
LOAD CSV WITH HEADERS FROM "file:///new_node.csv" AS line CREATE (:NewNode { title: line.title })

# 导入关系和属性
LOAD CSV WITH HEADERS FROM "file:///relation.csv" AS line
MATCH (entity1:EntityItem{title:line.Entity}), (entity2:NewNode{title:line.Newnode})
CREATE (entity1)-[:RELATION {type:line.relation}]->(entity2)
